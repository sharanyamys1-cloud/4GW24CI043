C Implementation of Producer–Consumer

The following C program implements the Producer–Consumer problem using a simple simulation approach. 

#include <stdio.h>
#include <stdlib.h>

#define SIZE 5

int buffer[SIZE];
int count = 0;

void produce() {
    if (count == SIZE) {
        printf("Buffer is FULL\n");
    } else {
        buffer[count++] = 1;
        printf("Item produced\n");
    }
}

void consume() {
    if (count == 0) {
        printf("Buffer is EMPTY\n");
    } else {
        count--;
        printf("Item consumed\n");
    }
}

int main() {
    int choice;
    while (1) {
        printf("1. Produce 2. Consume 3. Exit\n");
        scanf("%d", &choice);
        if (choice == 1) produce();
        else if (choice == 2) consume();
        else break;
    }
}
